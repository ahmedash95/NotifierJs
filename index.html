OPTIONS & DEFAULTS
Option Description Code
Title Set Notifier Title 
title : 'I am a Title'
Body Set Notifier Content 
body : 'am a notifier Content'
Icon Set Notifier Image 
icon : 'Path or URL of Icon\Image File'
TimeOut Set Norifire Time per Secound 
timeOut: 1; // 1 second

CALLBACKS
Function Code
Show 
show : Function(){ alert('show'); }
Click 
click : Function(){ alert('Click'); }
Close 
close : Function(){ alert('Close'); }
Error 
error : Function(){ alert('Error'); }

Default

$('#default').click(function(){
	notifi({
	title: 'Notifier ',
	icon: 'icon.png',
	body: 'Notifier',
	});
});

With Timer

$('#timer').click(function(){
  notifi({
    title: 'Notifier ',
    icon: 'icon.png',
    body: 'it will close After 3 Seconds',
    timeOut: 3,
  });
});

OnShow

$('#show').click(function(){
  notifi({
    title: 'Notifier ',
    icon: 'icon.png',
    body: 'Notifier',
    show: function(){
      alert('Hi i\'m Showing Now :D');
    }
  });
});

onClick

$('#click').click(function(){
  notifi({
    title: 'Notifier ',
    icon: 'icon.png',
    body: 'Notifier',
    click: function(){
      alert('You Are Clicked Me :)');
    }
  });
});

OnClose


$('#close').click(function(){
  notifi({
    title: 'Notifier ',
    icon: 'icon.png',
    body: '',
    close: function(){
      alert('Your Are Closed Me :\'( ');
    }
  });
});

OnError

$('#error').click(function(){
  notifi({
    title: 'Notifier ',
    icon: 'icon.png',
    body: '',
    error: function(){
      alert('Sorry For the Error :( ');
    }
  });
});
